import Episode, { frontmatter as metadata } from "./intro.mdx";

export default function Page() {
  console.log(metadata);
  const { audio, tags, guests, published_at } = metadata

  const tag = tags.map((tag) => {
    return <li key={tag}>{tag}</li>
  })

  const guest = Object.entries(guests).map(([name, link]) => {
    return (
      <li key={link}><a href={link}>{name}</a></li>
    )
  })

  const components = {
    h1: function ({ children }) {
      return (
        <>
          <h1>{children}</h1>
          <ul>{tag}</ul>
          <p>published_at: {published_at}</p>
          <audio controls={true} src={audio} />
          <ul>{guest}</ul>
        </>
      );
    }
  };
  return <Episode components={components} />;
}
